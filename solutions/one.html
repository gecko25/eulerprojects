Find the sum of all the multiples of 3 or 5 below 1000.

{{problemOneCtrl.answer}} 
 var sum = 0; // keep track of "passed" variables

//loop through all numbers 1-1000
	for (var i=1; i<1000; i++){

	if (i%3===0){
		sum+=i;
		continue;

	}else if (i%5===0){
		sum+=i;
		continue;

	}else{
		continue;
	}
}
